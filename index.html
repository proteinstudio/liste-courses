function lookupDB(name){

  var n=nrm(name);

  for(var i=0;i<DB_KEYS.length;i++){

    if(n.indexOf(nrm(DB_KEYS[i]))!==-1) return {key:DB_KEYS[i],d:DB[DB_KEYS[i]]};

  }

  return null;

}

function parseMult(qtyStr,itemName){

  if(!qtyStr||!qtyStr.trim()) return {mult:1,label:''};

  var s=qtyStr.trim().toLowerCase().replace(',','.');

  var e=lookupDB(itemName); var refG=e?e.d[3]:null;

  var massM=s.match(/^(\d+(?:\.\d+)?)\s*(g|kg|mg)$/);

  if(massM){

    var v=parseFloat(massM[1]),u=massM[2];

    var g=u==='kg'?v*1000:u==='mg'?v/1000:v;

    if(refG&&refG>0){var u2=Math.max(1,Math.ceil(g/refG));return {mult:u2,label:g+'g \u2192 \xd7'+u2};}

    return {mult:1,label:''};

  }

  var volM=s.match(/^(\d+(?:\.\d+)?)\s*(ml|cl|dl|l)$/);

  if(volM){

    var v2=parseFloat(volM[1]),u3=volM[2];

    var ml=u3==='l'?v2*1000:u3==='dl'?v2*100:u3==='cl'?v2*10:v2;

    if(refG&&refG>0){var u4=Math.max(1,Math.ceil(ml/refG));return {mult:u4,label:ml+'ml \u2192 \xd7'+u4};}

    return {mult:1,label:''};

  }

  var numM=s.match(/^x?(\d+(?:\.\d+)?)$/);

  if(numM){var n2=Math.ceil(parseFloat(numM[1]));if(!isNaN(n2)&&n2>0&&n2<=200)return {mult:n2,label:n2>1?'\xd7'+n2:''};}

  return {mult:1,label:''};

}

function calcPrice(item){

  var e=lookupDB(item.name); if(!e) return null;

  var pm=parseMult(item.qty,item.name);

  return {lo:e.d[0]*pm.mult,hi:e.d[1]*pm.mult,unit:e.d[2],mult:pm.mult,label:pm.label};

}

function computeBudget(){

  var pending=items.filter(function(x){return !x.done;});

  if(!pending.length) return null;

  var lo=0,hi=0,found=0,unknown=[];

  pending.forEach(function(item){

    var p=calcPrice(item);

    if(p){lo+=p.lo;hi+=p.hi;found++;}

    else unknown.push(item.name);

  });

  return {lo:lo,hi:hi,found:found,total:pending.length,unknown:unknown};

}

function renderBudget(){

  var b=computeBudget();

  var el=document.getElementById('budget');

  if(!b||b.total===0){el.classList.add('hide');return;}

  el.classList.remove('hide');

  document.getElementById('b-range').textContent=

    (b.lo===0&&b.hi===0)?'Non estimable':fmt(b.lo)+' \u20ac \u2013 '+fmt(b.hi)+' \u20ac';

  var pct=Math.round(b.found/b.total*100);

  var det=b.found+'/'+b.total+' articles estim\xe9s ('+pct+'%)';

  if(b.unknown.length>0&&b.unknown.length<=3) det+=' \xb7 non estim\xe9s\u00a0: '+b.unknown.join(', ');

  else if(b.unknown.length>3) det+=' \xb7 '+b.unknown.length+' non estim\xe9s';

}

function parseLine(line){

  line=line.replace(/^[\-\u2022*\u00b7\u2013\u2014>]+\s*/,'').replace(/[\u2018\u2019\u02bc']/g,' ').replace(/\(.*?\)/g,'').trim();

  if(!line) return null;

  var qty='';

  var qm=line.match(/^(\d[\d.,\/]*\s*(?:g|kg|ml|l|cl|dl)?)\s+/i);

  if(qm){qty=qm[1].trim();line=line.slice(qm[0].length);}

  var n=nrm(line);

  ARTCL.slice().sort(function(a,b){return b.length-a.length;}).forEach(function(a){n=n.replace(new RegExp('^'+nrm(a)+'\\s+'),'');});

  for(var ci=0;ci<CONT.length;ci++){var cm=n.match(new RegExp('(^|\\s)'+nrm(CONT[ci])+'s?\\s+(d |de |du |de la |des |d )?'));if(cm){var rest=n.slice(cm.index+cm[0].length).trim();if(rest){n=rest;break;}}}

  ARTCL.slice().sort(function(a,b){return b.length-a.length;}).forEach(function(a){n=n.replace(new RegExp('^'+nrm(a)+'\\s+'),'');});

  var words=n.split(/\s+/).filter(function(w){return NOISE.indexOf(w)===-1&&ARTCL.indexOf(w)===-1&&w.length>0;});

  var name=words.join(' ').trim();

  if(!name) return null;

  name=name.charAt(0).toUpperCase()+name.slice(1);

  return {name:name,qty:qty};

}
